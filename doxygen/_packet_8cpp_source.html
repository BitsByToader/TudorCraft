89//MARK: - Player Info
90void PlayerInfoPacket::read(TCPStream *stream) {
91 MCP::VarInt playerCount;
92 MCP::VarInt propertyCount;
93
94 *stream >> &m_action;
95 *stream >> &playerCount;
96
97 m_uuids.reserve(playerCount.value());
98
99 switch (m_action.value()) {
100 case 0: { // Add player
101 m_players.reserve(playerCount.value());
102 for ( int i = 0; i < playerCount.value(); i++ ) {
103 MCP::Uuid uuid;
104 *stream >> &uuid;
105 m_uuids.push_back(uuid);
106
107 Player player;
108
109 *stream >> &player.name;
110
111 *stream >> &propertyCount;
112 for ( int j = 0; j < propertyCount.value(); j++ ) {
113 Property property;
114
115 *stream >> &property;
116 player.properties.push_back(property);
117 }
118
119 *stream >> &player.gamemode;
120 *stream >> &player.ping;
121 *stream >> (int8_t *) &player.hasDisplayName;
122
123 if ( player.hasDisplayName )
124 *stream >> &player.displayName;
125
126 *stream >> (int8_t *) &player.hasSigData;
127 if ( player.hasSigData ) {
128 *stream >> &player.timestamp;
129 *stream >> &player.keyLength;
130 for ( int j = 0; j < player.keyLength.value(); j++ ) {
131 int8_t c;
132 *stream >> &c;
133 player.key.push_back(c);
134 }
135
136 *stream >> &player.signatureLength;
137 for ( int j = 0; j < player.signatureLength.value(); j++ ) {
138 int8_t c;
139 *stream >> &c;
140 player.signature.push_back(c);
141 }
142 }
143
144 m_players.push_back(player);
145 }
146
147 break;
148 }
149
150 case 1: { // Update gamemode
151 MCP::Uuid uuid;
152 *stream >> &uuid;
153 m_uuids.push_back(uuid);
154
155 m_gamemodes.reserve(playerCount.value());
156 for ( int i = 0; i < playerCount.value(); i++ ) {
157 MCP::Uuid uuid;
158 *stream >> &uuid;
159 m_uuids.push_back(uuid);
160
161 MCP::VarInt gm;
162 *stream >> &gm;
163 m_gamemodes.push_back(gm);
164 }
165
166 break;
167 }
168
169 case 2: { // Update latency
170 m_gamemodes.reserve(playerCount.value());
171 for ( int i = 0; i < playerCount.value(); i++ ) {
172 MCP::Uuid uuid;
173 *stream >> &uuid;
174 m_uuids.push_back(uuid);
175
176 MCP::VarInt ping;
177 *stream >> &ping;
178 m_pings.push_back(ping);
179 }
180
181 break;
182 }
183
184 case 3: { // Update display name
185 m_names.reserve(playerCount.value());
186 for ( int i = 0; i < playerCount.value(); i++ ) {
187 MCP::Uuid uuid;
188 *stream >> &uuid;
189 m_uuids.push_back(uuid);
190
191 NewDisplayName name;
192 *stream >> (int8_t *) &name.hasDisplayName;
193 if ( name.hasDisplayName )
194 *stream >> &name.displayName;
195
196 m_names.push_back(name);
197 }
198 break;
199 }
200 case 4: { // Remove player
201 for ( int i = 0; i < playerCount.value(); i++ ) {
202 MCP::Uuid uuid;
203 *stream >> &uuid;
204 m_uuids.push_back(uuid);
205 }
206 break;
207 }
208
209 default:
210 break;
211 }

301 *stream >> &m_chunkX;
302 *stream >> &m_chunkZ;

304 *stream >> &m_heightmap;
305 *stream >> &m_dataSize;

307 for ( int chunkY = 0; chunkY < 384/16; chunkY++ ) {
308 ChunkSection section;
309
310 *stream >> &section.blockCount;
311
312 // BlockStates Pallete

315 if ( section.blockStates.bitsPerEntry == 0 ) {

369void ChunkDataPacket::updateGame() {
370 simd::float2 pos = { (float)m_chunkX, (float)m_chunkZ };
371 simd::float2 center = World::shared()->centerChunk;
372
373 if ( simd::distance(center, pos) > 3 )
374 return;
375
376 std::unordered_map<int, BlockState*> pallete = BlockState::GlobalPallete();
377 World *w = World::shared();
378
379 Renderer::shared()->m_gpuMutex.lock();
380
381 // For every chunk in the section
382 for ( int chunkY = 0; chunkY < 384/16; chunkY++ ) {
383 MCP::ChunkDataPacket::ChunkSection section = m_data.at(chunkY);
384 MCP::ChunkDataPacket::ChunkSection::PalletedContainer blocks = section.blockStates;
385
386 Chunk *newChunk = new Chunk(m_chunkX, chunkY-4, m_chunkZ);
387 w->loadChunk(newChunk);
388
389 int blockCount = section.blockCount;
390
391 if ( blocks.bitsPerEntry == 0 || blockCount == 0 ) {
392 continue;
393 }
394
395 int longCount = 0;
396 int longShiftCount = 0;
397 int64_t longToUse = blocks.dataArray[0];
398 int64_t mask = pow(2, blocks.bitsPerEntry) - 1;
399
400 // For every block in the chunk
401 for ( int y = 0; y < 16; y++ ) {
402 for ( int z = 0; z < 16; z++ ) {
403 for ( int x = 0; x < 16; x++ ) {
404 int64_t indexInPallete = longToUse & mask;
405
406 longToUse >>= blocks.bitsPerEntry;
407
408 longShiftCount++;
409
410 if ( longShiftCount == 64/blocks.bitsPerEntry ) {
411 longShiftCount = 0;
412 longCount++;
413 longToUse = blocks.dataArray[longCount];
414 }
415
416 try {
417 newChunk->placeBlockAt(x, y, z,
418 pallete.at(blocks.pallete[indexInPallete].value()));
419 } catch(...) {
420 newChunk->placeBlockAt(x, y, z, BlockState::UndefinedBlock());
421 }
422 }
423 }
424 }
425 }
426
427 Renderer::shared()->m_gpuMutex.unlock();
428};
