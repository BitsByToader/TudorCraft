22simd::float3 PhysicsConformable::calculateFinalVelocity() {
23 simd::float3 finalV = { 0.f, 0.f, 0.f };
24
25 int sign = 1;
26 for ( auto &i: m_velocities ) {
27 if ( i.second.x == 0 && i.second.y == 0 && i.second.z == 0 ) {
28 continue;
29 }
30
31 // Calculate drag for velocity
32 simd::float3 drag = simd::normalize(i.second);
33 drag *= m_dragMagnitude;
34
35 // Check if we're chaning direction
36 float copyX = i.second.x;
37 sign = copyX < 0 ? -1 : 1;
38 copyX *= sign; // Make the component positive
39 copyX -= (drag.x < 0 ? -drag.x : drag.x); // Apply drag
40 if ( copyX < 0 ) {
41 i.second = { 0.f, 0.f, 0.f };
42 } else {
43 // Apply drag
44 i.second -= drag;
45 }
46
47 finalV += i.second;
48 }
49
50 return finalV;
51};
