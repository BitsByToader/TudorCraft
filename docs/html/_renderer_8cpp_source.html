263//MARK: - Create Heap
264void Renderer::createHeap() {
265 NS::AutoreleasePool* pPool = NS::AutoreleasePool::alloc()->init();
266
267 MTL::HeapDescriptor *heapDescriptor = MTL::HeapDescriptor::alloc()->init();
268 heapDescriptor->setStorageMode(MTL::StorageModePrivate);
269 heapDescriptor->setSize(0);
270
271 //Build a descriptor for each texture and calculate the size required to store all the textures in the heap.
272 for ( int i = 0; i < TEXTURE_COUNT; i++ ) {
273 MTL::TextureDescriptor *descriptor = Renderer::newDescriptorFromTexture(m_texture[i],
274 heapDescriptor->storageMode());
275
276 MTL::SizeAndAlign sizeAndAlign = m_device->heapTextureSizeAndAlign(descriptor);
277 sizeAndAlign.size += (sizeAndAlign.size & (sizeAndAlign.align - 1)) + sizeAndAlign.align;
278 heapDescriptor->setSize(heapDescriptor->size() + sizeAndAlign.size);
279
280 descriptor->release();
281 }
282
283 m_heap = m_device->newHeap(heapDescriptor);
284
285 heapDescriptor->release();
286 pPool->release();
287};

289//MARK: - Move resources to heap
290void Renderer::moveResourcesToHeap() {
291 NS::AutoreleasePool *pPool = NS::AutoreleasePool::alloc()->init();
292
293 // Create a command buffer and blit encoder to copy data from existing resources
294 // to resources created from the heap.
295 MTL::CommandBuffer *commandBuffer = m_commandQueue->commandBuffer();
296 commandBuffer->setLabel(AAPLSTR("Heap copy command buffer"));
297
298 MTL::BlitCommandEncoder *blitEncoder = commandBuffer->blitCommandEncoder();
299 blitEncoder->setLabel(AAPLSTR("Heap transfer blit encoder"));
300
301 // Create new textures from the heap and copy the contents.
302 for ( int i = 0; i < TEXTURE_COUNT; i++ ) {
303 MTL::TextureDescriptor *descriptor = Renderer::newDescriptorFromTexture(m_texture[i],
304 m_heap->storageMode());
305
306 MTL::Texture *heapTexture = m_heap->newTexture(descriptor);
307 heapTexture->setLabel(m_texture[i]->label());
308
309 MTL::Region region = MTL::Region::Make2D(0, 0, m_texture[i]->width(), m_texture[i]->height());
310 for ( int level = 0; level < m_texture[i]->mipmapLevelCount(); level++ ) {
311 for ( int slice = 0; slice < m_texture[i]->arrayLength(); slice++ ) {
312 blitEncoder->copyFromTexture(m_texture[i], // sourceTexture
313 slice, // sourceSlice
314 level, // sourceLevel
315 region.origin, // sourceOrigin
316 region.size, // sourceSize
317 heapTexture, // toTexture
318 slice, // destinationSlice
319 level, // destinationLevel
320 region.origin); // destinationTexture
321 }
322
323 region.size.width /= 2;
324 region.size.height /= 2;
325 if(region.size.width == 0) region.size.width = 1;
326 if(region.size.height == 0) region.size.height = 1;
327 }
328
329 descriptor->release();
330 m_texture[i]->release();
331 m_texture[i] = heapTexture;
332 }
333
334 blitEncoder->endEncoding();
335 commandBuffer->commit();
336
337 pPool->release();
338};

352// MARK: - Draw method
353void Renderer::draw(std::vector<std::shared_ptr<Entity>>& entities,
354 MTL::RenderPassDescriptor *currentRPD,
355 MTL::Drawable* currentDrawable) {
356 if ( m_instanceCount == 0 ) {
357 return;
358 }
359
360 NS::AutoreleasePool* pPool = NS::AutoreleasePool::alloc()->init();
361
362 using namespace simd;
363
364 // Create a new command buffer for each render pass to the current drawable.
365 MTL::CommandBuffer *commandBuffer = m_commandQueue->commandBuffer();
366 commandBuffer->setLabel(AAPLSTR("WorldDrawingCommand"));
367
368 // Wait for the instance buffer to be available
369 m_gpuMutex.lock();
370 commandBuffer->addCompletedHandler( ^void( MTL::CommandBuffer* pCmd ) {
371 m_gpuMutex.unlock();
372 });
373
374 int cubeCount = 0;
375 InstanceData *cubes = reinterpret_cast<InstanceData *>(m_cubeInstanceDatBuffer->contents());
376 for ( auto entity: entities ) {
377 for ( auto component: entity->components() ) {
378 cubes[cubeCount].transform = entity->transformMatrix() * component->transformMatrix();
379 cubes[cubeCount].normalTransform = Math3D::discardTranslation(cubes[cubeCount].transform);
380 cubes[cubeCount].textureId = 3;
381 cubes[cubeCount].highlighted = false;
382
383 cubeCount++;
384 }
385 }
386
387 // MARK: World transformations
388 CameraData *cameraData = reinterpret_cast<CameraData *>(m_cameraDataBuffer->contents());
389 cameraData->worldTranform = Math3D::makeXRotate4x4(-m_cameraPitch) * Math3D::makeYRotate4x4(-m_cameraYaw) * Math3D::makeTranslate( -m_cameraPosition );
390 cameraData->worldNormalTranform = Math3D::discardTranslation(cameraData->worldTranform);
391
392 // MARK: Configure draw command
393 if ( currentRPD != nullptr ) {
394 // Create a render command encoder.
395 MTL::RenderCommandEncoder *renderEncoder = commandBuffer->renderCommandEncoder(currentRPD);
396
397 // Set the region of the drawable to draw into.
398 MTL::Viewport viewPort = {0.0, 0.0, (double) m_windowSize.x, (double) m_windowSize.y, 0.0, 1.0};
399 renderEncoder->setViewport(viewPort);
400
401 // Indicate to Metal that a GPU heap will be used.
402 renderEncoder->useHeap(m_heap);
403
404 renderEncoder->setRenderPipelineState(m_pipelineState);
405 renderEncoder->setDepthStencilState(m_depthState);
406
407 // Pass in the parameter data for cube faces.
408 renderEncoder->setVertexBuffer(m_blockFaceVertices, 0, VertexInputIndexVertices);
409 renderEncoder->setVertexBuffer(m_instanceDataBuffer, 0, VertexInputIndexInstanceData);
410 renderEncoder->setVertexBuffer(m_cameraDataBuffer, 0, VertexInputIndexCameraData);
411 renderEncoder->setFragmentBuffer(m_fragmentShaderArgBuffer, 0, FragmentInputIndexTextures);
412 renderEncoder->setFragmentBuffer(m_cameraDataBuffer, 0, FragmentInputIndexCameraData);
413
414 // Set modes for 3D rendering
415 renderEncoder->setCullMode( MTL::CullModeBack );
416 renderEncoder->setFrontFacingWinding( MTL::Winding::WindingCounterClockwise );
417
418 // Draw our cube faces
419 renderEncoder->drawIndexedPrimitives(MTL::PrimitiveType::PrimitiveTypeTriangle, // Primitive type
420 6, // Index count
421 MTL::IndexType::IndexTypeUInt16, // Index type
422 m_blockFaceIndexBuffer, // Index buffer
423 0, // Index buffer offset
424 m_instanceCount); // Instance count
425
426 // Reencode paramaters for cube drawing
427 renderEncoder->setVertexBuffer(m_cubeVertices, 0, VertexInputIndexVertices);
428 renderEncoder->setVertexBuffer(m_cubeInstanceDatBuffer, 0, VertexInputIndexInstanceData);
429
430 // Draw our cubes
431 renderEncoder->drawIndexedPrimitives(MTL::PrimitiveType::PrimitiveTypeTriangle, // Primitive type
432 36, // Index count
433 MTL::IndexType::IndexTypeUInt16, // Index type
434 m_cubeIndexBuffer, // Index buffer
435 0, // Index buffer offset
436 cubeCount); // Instance count
437
438 renderEncoder->endEncoding();
439 commandBuffer->presentDrawable(currentDrawable);
440 }
441
442 commandBuffer->commit();
443
444 pPool->release();
445}

457//MARK: - Remove instance method
458void Renderer::removeInstanceAt(int index) {
459 InstanceData *buffer = reinterpret_cast<InstanceData *>(m_instanceDataBuffer->contents());
460 buffer[index] = buffer[m_instanceCount - 1];
461
462 // Get the block we moved the instance of
463 Block *b = World::shared()->getBlockAt(buffer[index].blockCoordinates.x,
464 buffer[index].blockCoordinates.y,
465 buffer[index].blockCoordinates.z);
466
467 // Search for the face with the old index and update it to the new index.
468 if ( b != nullptr ) {
469 for ( int i = 0; i < 6; i++ ) {
470 if ( b->faceIndices[i] == m_instanceCount-1 ) {
471 b->faceIndices[i] = index;
472 break;
473 }
474 }
475 }
476
477 --m_instanceCount;
478};
