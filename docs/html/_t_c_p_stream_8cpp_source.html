35TCPStream::TCPStream(MCP::ConnectionState *state, std::string serverAddress, int port) {
36 m_connectionState = state;
37
38 memset(m_sendBuffer, 0x0, RECV_BUFFER_SIZE);
39 memset(m_sendBuffer, 0x0, SEND_BUFFER_SIZE);
40
41 // Configure the TCP socket
42 m_socketDescriptor = socket(AF_INET, SOCK_STREAM, 0);
43 if ( m_socketDescriptor == -1 ) {
44 AAPL_PRINT("Couldn't create TCP socket!");
45 return;
46 }
47
48 struct sockaddr_in addr;
49 addr.sin_family = AF_INET;
50 addr.sin_port = htons(port);
51 if ( inet_pton(AF_INET, serverAddress.c_str(), &addr.sin_addr) <= 0 ) {
52 AAPL_PRINT("Server address is invalid!");
53 return;
54 }
55
56 if ( connect(m_socketDescriptor, (struct sockaddr *)&addr, sizeof(addr)) < 0 ) {
57 AAPL_PRINT("Couldn't connect to server!");
58 return;
59 }
60
61 AAPL_PRINT("Connected to TCP server!");
62};

89size_t TCPStream::readBytes(uint8_t *buffer, size_t length) {
90 int64_t read = 0;
91
92 while ( length > 0 ) {
93 int64_t biggestSizePossible = (length > RECV_BUFFER_SIZE) ? RECV_BUFFER_SIZE : length;
94 int64_t bytesRead = recv(m_socketDescriptor, buffer+read, biggestSizePossible, 0);
95
96 m_currentPacketBytesRead += bytesRead;
97 length -= bytesRead;
98 read += bytesRead;
99
100 if ( bytesRead == 0 || bytesRead == -1 ) {
101 *m_connectionState = MCP::ConnectionState::Unconnected;
102 return -1;
103 }
104 }
105
106 return read;
107};

383 int readValue = 0;
384 unsigned char currentByte = 0;
385
386 while ( true ) {
387 readBytes(&currentByte, 1); // Read current byte
388 readValue |= ( currentByte & SEGMENT_BITS ) << position*7;
