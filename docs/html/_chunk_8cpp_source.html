44void Chunk::calculateMesh(Renderer *renderer) {
45 using namespace Math3D;
46 using namespace simd;
47
48 const float scl = BLOCK_SIZE;
49
50 int *instanceCount = renderer->instanceCount();
51 InstanceData *buffer = renderer->instanceBuffer();
52
53 const float4x4 scale = makeScale( (float3){ scl, scl, scl } );
54
55 World *w = World::shared();
56 Block *nextBlock;
57
58 for ( int k = 0; k < BLOCKS_NUM_Y; k++ ) {
59 for ( int j = 0; j < BLOCKS_NUM_Z; j++ ) {
60 for ( int i = 0; i < BLOCKS_NUM_X; i++ ) {
61 Block *currBlock = getBlockAt(i, j, k);
62
63 // Only check blocks that aren't air
64 if ( currBlock->state != nullptr ) {
65 m_blockCount++;
66
67 int x = m_xCoordinate * BLOCKS_NUM_X + i;
68 int y = m_yCoordinate * BLOCKS_NUM_Y + k;
69 int z = m_zCoordinate * BLOCKS_NUM_Z - j;
70
71 // Create the main block position
72 float4x4 blockPositionTranslationMatrix =
73 makeTranslate((float3) { x*scl, y*scl, -z*scl} );
74
75 nextBlock = w->getBlockAt(x, y, z-1);
76 if ( nextBlock == nullptr || nextBlock->state == nullptr ) {
77 // Front of block
78 buffer[*instanceCount].transform = blockPositionTranslationMatrix * scale;
79 buffer[*instanceCount].normalTransform = Math3D::discardTranslation(buffer[*instanceCount].transform);
80 buffer[*instanceCount].textureId = currBlock->state->frontTexture();
81 currBlock->faceIndices[Faces::Front] = *instanceCount;
82
83 (*instanceCount)++;
84 }
85
86 nextBlock = w->getBlockAt(x-1, y, z);
87 if ( nextBlock == nullptr || nextBlock->state == nullptr ) {
88 // Left of block
89 buffer[*instanceCount].transform = blockPositionTranslationMatrix * scale * moveFaceToLeft();
90 buffer[*instanceCount].normalTransform = discardTranslation(buffer[*instanceCount].transform);
91 buffer[*instanceCount].textureId = currBlock->state->leftTexture();
92 currBlock->faceIndices[Faces::Left] = *instanceCount;
93
94 (*instanceCount)++;
95 }
96
97 nextBlock = w->getBlockAt(x, y, z+1);
98 if ( nextBlock == nullptr || nextBlock->state == nullptr ){
99 // Back of block
100 buffer[*instanceCount].transform = blockPositionTranslationMatrix * scale * moveFaceToBack();
101 buffer[*instanceCount].normalTransform = discardTranslation(buffer[*instanceCount].transform);
102 buffer[*instanceCount].textureId = currBlock->state->backTexture();
103 currBlock->faceIndices[Faces::Back] = *instanceCount;
104
105 (*instanceCount)++;
106 }
107
108 nextBlock = w->getBlockAt(x+1, y, z);
109 if ( nextBlock == nullptr || nextBlock->state == nullptr ){
110 // Right of block

112 buffer[*instanceCount].normalTransform = discardTranslation(buffer[*instanceCount].transform);

146void Chunk::placeBlockAt(int x, int y, int z, BlockState *state, Renderer *renderer, World *world) {
147 using namespace simd;
148 using namespace Math3D;
149
150 InstanceData *buffer = renderer->instanceBuffer();
151 int *count = renderer->instanceCount();
152
153 Block *currBlock = getBlockAt(x, y, z);
154 if ( currBlock->state == nullptr && state == nullptr )
155 return;
156
157 if ( currBlock->state != nullptr && state != nullptr ) {
158 /*
159 A hack to update an already existing block with a new state.
160 I initially made the method to basically place or remove blocks... and that's it.
161 To change a block's state, I'd need to add a whole other edge case where we don't
162 remove any instances but only update their textures (since that's what the state is about
163 to the renderer). This hack isn't really noticeable to the naked eye, so it's easier to
164 go about it this way.
165 */
166 placeBlockAt(x, y, z, nullptr); // Remove the block
167 placeBlockAt(x, y, z, state); // Place a block with the new state
168 return;
169 }
170
171 currBlock->state = state;
172
173 Block *nextBlock = nullptr;
174
175 const float scl = BLOCK_SIZE;
176 const float4x4 scale = makeScale( (float3){ scl, scl, scl } );
177
178 int globalX = m_xCoordinate * BLOCKS_NUM_X + x;
179 int globalY = m_yCoordinate * BLOCKS_NUM_Y + y;
180 int globalZ = m_zCoordinate * BLOCKS_NUM_Z + z;
181
182 if ( state != nullptr ) {
183 // We're placing a block.
184 m_blockCount++;
185
186 float4x4 blockPositionTranslationMatrix =
187 makeTranslate((float3) { globalX*scl, globalY*scl, -globalZ*scl} );
188
189
190 // Front of block
191 nextBlock = world->getBlockAt(globalX, globalY, globalZ-1);
192 if ( nextBlock == nullptr || nextBlock->state == nullptr ) {
193 // The block doesn't exist or it is an air block.
194 // Draw face.
195 buffer[*count].transform = blockPositionTranslationMatrix * scale;
196 buffer[*count].normalTransform = Math3D::discardTranslation(buffer[*count].transform);
197 buffer[*count].textureId = currBlock->state->frontTexture();
198 buffer[*count].blockCoordinates = {globalX, globalY, globalZ};
199 currBlock->faceIndices[Faces::Front] = *count;
200
201 ++(*count);
202
203 } else {
204 // The block exists and it isn't an air block.
205 // Don't draw face, but remove the neighbour's face.
206 renderer->removeInstanceAt(nextBlock->faceIndices[Faces::Back]);
207 nextBlock->faceIndices[Faces::Back] = -1;
208 }
209
210 // Left of block
211 nextBlock = world->getBlockAt(globalX-1, globalY, globalZ);
212 if ( nextBlock == nullptr || nextBlock->state == nullptr ) {
213 buffer[*count].transform = blockPositionTranslationMatrix * scale * moveFaceToLeft();
214 buffer[*count].normalTransform = discardTranslation(buffer[*count].transform);
215 buffer[*count].textureId = currBlock->state->leftTexture();
216 buffer[*count].blockCoordinates = {globalX, globalY, globalZ};
217 currBlock->faceIndices[Faces::Left] = *count;
218
219 ++(*count);
220 } else {
221 renderer->removeInstanceAt(nextBlock->faceIndices[Faces::Right]);
222 nextBlock->faceIndices[Faces::Right] = -1;
223 }
224
