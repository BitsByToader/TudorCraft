56void Engine::updateGame(MTL::RenderPassDescriptor *currentRPD, MTL::Drawable* currentDrawable) {
57 auto now = std::chrono::high_resolution_clock::now();
58 std::chrono::duration<double, std::milli> passed = now - lastUpdate;
59
60 for(auto &i: m_entities) {
61 i.second->updatePosition(passed.count());
62 }
63
64 if ( m_player ) {
65#warning This packet should normally be sent every 50ms (a server tick), but this works as well.
66 MCP::SetPlayerPositionAndRotation packet;
67 packet.m_x = m_player->position().x / 10;
68 packet.m_y = m_player->position().y / 10;
69 packet.m_z = -m_player->position().z / 10;
70 packet.m_yaw = -m_player->rotation().y / M_PI * 180.f - 90.f;
71 packet.m_pitch = m_player->cameraBoundComponent->rotation().x / M_PI * 180.f;
72 packet.m_onGround = m_player->onGround;
73
74 m_serverConnection->sendPacket(&packet);
75 }
76
77 lastUpdate = now;
78
79 checkInput();
80
81 updateCamera();
82
83 updateLookTarget();
84
85 std::vector<std::shared_ptr<Entity>> entities;
86 for ( auto &i: m_entities ) {
87 entities.push_back(i.second);
88 }
89
90 m_renderer->draw(entities, currentRPD, currentDrawable);
91};

93void Engine::updateLookTarget() {
94 if ( !m_player)
95 return;
96
97 float pitch = m_player->cameraBoundComponent->rotation().x;
98 float yaw = m_player->rotation().y;
99 simd::float3 rayPosition = m_player->cameraPosition();
100
101 simd::float3 lookDirection = {-cos(pitch) * sin(yaw),
102 -sin(pitch),
103 -cos(pitch) * cos(yaw)};
104
105 for ( float i = 0; i < 50; i += 0.1 ) {
106 // Only highlight a block if it isn't more than 5 block away
107 simd::int3 blockToCheck = { (int)rayPosition.x / 10,
108 (int)rayPosition.y / 10,
109 (int)-rayPosition.z / 10
110 };
111
112 Block *b = m_world->getBlockAt(blockToCheck.x, blockToCheck.y, blockToCheck.z);
113 if ( b != nullptr && b->state != nullptr ) {
114 // We found a non air block to highlight!
115 if ( m_highlightedBlockCoords.x != blockToCheck.x ||
116 m_highlightedBlockCoords.y != blockToCheck.y ||
117 m_highlightedBlockCoords.z != blockToCheck.z)
118 {
119 m_world->highlightBlock(blockToCheck.x, blockToCheck.y, blockToCheck.z);
120 m_highlightedBlockCoords = blockToCheck;
121 m_raycastHitPosition = rayPosition;
122 }
123
124 break;
125 }
126
127 // Didn't find block to highlight
128// m_world->removeHightlight();
129
130 // Go through the entities here and check their position when implementing hits.
131 // Important to check after the blocks as to first remove a block and then hit the entity behind it.
132
133 rayPosition += lookDirection * 0.1;
134 }

146void Engine::checkInput() {
147 if ( !m_player )
148 return;
149
150 float pitch = m_player->cameraBoundComponent->rotation().x;
151 float yaw = m_player->rotation().y;
152
153 // Check mouse
154 if ( m_yawRadiansLeft > 0 ) {
155 // Right
156 if (yaw == -2*M_PI ) {
157 yaw = 0;
158 } else {
159 yaw -= 0.05f;
160 }
161
162 m_yawRadiansLeft -= 0.5;
163 } else if ( m_yawRadiansLeft < 0 ){
164 // Left
165 if ( yaw == 2*M_PI ) {
166 yaw = 0;
167 } else {
168 yaw += 0.05f;
169 }
170
171 m_yawRadiansLeft += 0.5;
172 }
173
174 if ( m_pitchRadiansLeft < 0 ) {
175 // Up
176 if ( pitch > -M_PI/2 ) {
177 pitch -= 0.05f;
178 }
179
180 m_pitchRadiansLeft += 0.5;
181 } else if ( m_pitchRadiansLeft > 0 ) {
182 // Down
183 if ( pitch < M_PI/2) {
184 pitch += 0.05f;
185 }
186
187 m_pitchRadiansLeft -= 0.5;
188 }
189
190 simd::float3 moveDirection = {-7.5f * sin(yaw),
191 0.f,
192 -7.5f * cos(yaw)};
193
194 simd::float3 movementVector = { 0.f, 0.f, 0.f };
195
196 bool playerMoved = false;
197
198 // Check keyboard
199 if ( m_goingForward ) {
200 movementVector.x += moveDirection.x;
201 movementVector.z += moveDirection.z;
202
203 playerMoved = true;
204 }
205
206 if ( m_goingLeft ) {
207 movementVector.x += (moveDirection * Math3D::makeYRotate3x3(-M_PI/2)).x;
208 movementVector.z += (moveDirection * Math3D::makeYRotate3x3(-M_PI/2)).z;
209
210 playerMoved = true;
211 }
212
213 if ( m_goingRight ) {
214 movementVector.x += (moveDirection * Math3D::makeYRotate3x3(M_PI/2)).x;
215 movementVector.z += (moveDirection * Math3D::makeYRotate3x3(M_PI/2)).z;
216
217 playerMoved = true;
218 }
219
220 if ( m_goingBackward ) {
221 movementVector.x -= moveDirection.x;
222 movementVector.z -= moveDirection.z;
223
224 playerMoved = true;
225 }
226
227 if ( m_goingUp ) {
228 movementVector.y += 20.f;
229
230 playerMoved = true;
231 }
232
233 if ( playerMoved && m_player->onGround )
234 m_player->setSpeed(movementVector.x, movementVector.y, movementVector.z);
235
236 simd::float3 temp = m_player->rotation();
237 m_player->setRotation(temp.x, yaw);
238
239 temp = m_player->cameraBoundComponent->rotation();
240 m_player->cameraBoundComponent->setRotation(pitch, temp.y, temp.z);
241};

354void Engine::pressedMouse(int button) {
355 m_renderer->m_gpuMutex.lock();
356 // 0 - left
357 // 1 - right
358 if ( button == 0 ) {
359 // Remove block
360 m_world->placeBlockAt(m_highlightedBlockCoords.x,
361 m_highlightedBlockCoords.y,
362 m_highlightedBlockCoords.z, nullptr);
363
364 MCP::PlayerActionPacket startdig(m_highlightedBlockCoords.x,
365 m_highlightedBlockCoords.y,
366 m_highlightedBlockCoords.z,
367 m_sequenceCounter);
368 m_serverConnection->sendPacket(&startdig);
369 } else if ( button == 1 ) {
370 // Place block
371 simd::float3 lookDirection = {
372 m_highlightedBlockCoords.x + .5f,
373 m_highlightedBlockCoords.y + .5f,
374 m_highlightedBlockCoords.z + .5f,
375 };
376 lookDirection -= m_raycastHitPosition;
377
378 const simd::float3 vecRight = {1.f, 0.f, 0.f};
379 const simd::float3 vecUp = {0.f, 1.f, 0.f};
380 const simd::float3 vecForward = {0.f, 0.f, 1.f};
381
382 float up = Math3D::cosBetween(lookDirection, vecUp);
383 float down = -up;
384 float right = Math3D::cosBetween(lookDirection, vecRight);
385 float left = -right;
386 float forward = Math3D::cosBetween(lookDirection, vecForward);
387 float backward = -forward;
388
389 float min = std::min({up, down, right, left, forward, backward});
390
391 int face = 0;
392 if ( min == up ) {
393 // Place on top
394 m_world->placeBlockAt(m_highlightedBlockCoords.x,
395 m_highlightedBlockCoords.y+1,
396 m_highlightedBlockCoords.z, BlockState::GrassBlock());
397 face = 1;
398
399 } else if ( min == down ) {
400 // Place on bottom
401 m_world->placeBlockAt(m_highlightedBlockCoords.x,
402 m_highlightedBlockCoords.y-1,
403 m_highlightedBlockCoords.z, BlockState::GrassBlock());
404 face = 0;
405
406 } else if ( min == left ) {
407 // Place to the left
408 m_world->placeBlockAt(m_highlightedBlockCoords.x-1,
409 m_highlightedBlockCoords.y,
410 m_highlightedBlockCoords.z, BlockState::GrassBlock());
411 face = 4;
412
413 } else if ( min == right ) {
414 // Place to the right
415 m_world->placeBlockAt(m_highlightedBlockCoords.x+1,
416 m_highlightedBlockCoords.y,
417 m_highlightedBlockCoords.z, BlockState::GrassBlock());
418 face = 5;
419
420 } else if ( min == forward ) {
421 // Place in front
422 m_world->placeBlockAt(m_highlightedBlockCoords.x,
423 m_highlightedBlockCoords.y,
424 m_highlightedBlockCoords.z-1, BlockState::GrassBlock());
425 face = 2;
426
427 } else if ( min == backward ) {
428 // Place in the back
429 m_world->placeBlockAt(m_highlightedBlockCoords.x,
430 m_highlightedBlockCoords.y,
431 m_highlightedBlockCoords.z+1, BlockState::GrassBlock());
432 face = 3;
433
434 }
435
436 m_world->removeHightlight();
437
438 MCP::UseItemOnPacket place(m_highlightedBlockCoords.x,
439 m_highlightedBlockCoords.y,
440 m_highlightedBlockCoords.z,
441 face,
442 m_sequenceCounter);
443 m_serverConnection->sendPacket(&place);
444 }
445 m_renderer->m_gpuMutex.unlock();
446 m_sequenceCounter++;
447};
