57MTL::Texture *TextureAtlas::getTextureWithCoordinates(int x, int y) {
58 for ( int i = 0; i < 16; i++ ) {
59 for ( int j = 0; j < 16; j++ ) {
60 // Swap red with blue channels
61 m_textureRawData[(i*16+j)*4 + 0] = m_atlasRawData[((x+i) * m_atlasHeight + (y+j)) * 4 + 2];
62 m_textureRawData[(i*16+j)*4 + 1] = m_atlasRawData[((x+i) * m_atlasHeight + (y+j)) * 4 + 1];
63 m_textureRawData[(i*16+j)*4 + 2] = m_atlasRawData[((x+i) * m_atlasHeight + (y+j)) * 4 + 0];
64 m_textureRawData[(i*16+j)*4 + 3] = m_atlasRawData[((x+i) * m_atlasHeight + (y+j)) * 4 + 3];
65 }
66 }
67
68 // Indicate that each pixel has a blue, green, red, and alpha channel, where each channel is
69 MTL::TextureDescriptor *textureDescriptor = MTL::TextureDescriptor::alloc()->init();
70 textureDescriptor->setPixelFormat(MTL::PixelFormatBGRA8Unorm_sRGB);
71
72 // Set the pixel dimensions of the texture
73 textureDescriptor->setWidth(TEXTURE_WIDTH);
74 textureDescriptor->setHeight(TEXTURE_HEIGHT);
75
76 NS::UInteger bytesPerRow = m_channels * TEXTURE_WIDTH;
77
78 // Create the texture from the device by using the descriptor
79 MTL::Texture *texture = m_device->newTexture(textureDescriptor);
80
81 MTL::Region region = MTL::Region(0, 0, 0, // MTLOrigin
82 16, 16, 1); // MTLSize
83
84 // Copy the bytes from the data object into the texture
85 texture->replaceRegion(region, 0, m_textureRawData, bytesPerRow);
86
87 textureDescriptor->release();
88
89 return texture;
90};
